### 功能性测试
- *VIK* (输出端在抽出规定电流时的电压)
- *VOH* (输入端施加规定电平，使输出为逻辑高电平时的电压)
- *VOL* (输入端施加规定电平，使输出为逻辑低电平时的电压)
- *II* (输入端施加规定电平电压$V_i$时流入器件的电流)
- *IIH* (输入端施加规定电平电压$V_{ih}$时流入器件的电流)
- *IIL* (输入端施加规定电平电压$V_{il}$时流入器件的电流)
- *IOS* (输入端施加规定的电平，使输出为逻辑高电平时输出端对地短路电流)
- *ICC* (输入端在施加规定电平，经电源端流入器件的电流)
- *ICCH*
- *ICCL*

参考程序
```cpp{.line-numbers}
SET_DPS(1,5,V,50,MA);
SET_INPUT_LEVEL(2,0.7);
SET_OUTPUT_LEVEL(2.5,0.4);
SET_PERIOD(2000);
SET_TIMING(200,1400,1700);
FORMAT(NRZ0,"1,2,4,5,44,43,47,46");
if(!RUN_PATTERN("FUN",0,1,0,0))
    BIN(2);

// VOH测试
SET_DPS(1,4.5,V,50,MA);
SET_INPUT_LEVEL(2,0.7);
RUN_PATTERN(2,1,0,0);
PMU_CONDITIONS(FIMV,-0.4,MA,5,V);
if(!PMU_MEASURE("3,6,42,45",5,"VOH_",V,No_UpLimit,2.5))
    BIN(4);

// VOL测试
SET_DPS(1,4.5,V,50,MA);
SET_INPUT_LEVEL(2,0.7);
RUN_PATTERN(1,1,0,0);
PMU_CONDITIONS(FIMV,4,MA,0.6,V);
if(!PMU_MEASURE("3,6,42,45",5,"VOL_",V,0.4,No_LoLimit))
    BIN(5);

// VIK测试
SET_DPS(1,4.5,V,50,MA);
PMU_CONDITIONS(FIMV,-18,MA,2,V);
if(!PMU_MEASURE("1,2,4,5,43,44,46,47",5,"VIK_",V,No_UpLimit,-1.5))
    BIN(3);

// II测试
SET_DPS(1,5.5,V,50,MA);
RUN_PATTERN(2,1,0,0);
PMU_CONDITIONS(FVMI,7,V,0.19,MA);
if(!PMU_MEASURE("1,2,4,5,43,44,46,47",5,"II_",MA,0.1,No_LoLimit))
    BIN(6);

// IIH测试
SET_DPS(1,5.5,V,50,MA);
RUN_PATTERN(2,1,0,0);
PMU_CONDITIONS(FVMI,2.7,V,50,UA);
if(!PMU_MEASURE("1,2,4,5,43,44,46,47",5,"IIH_",UA,20,No_LoLimit))
    BIN(6);

// IIL测试
SET_DPS(1,5.5,V,50,MA);
RUN_PATTERN(1,1,0,0);
PMU_CONDITIONS(FVMI,0.4,V,1,MA);
if(!PMU_MEASURE("1,2,4,5,43,44,46,47",5,"IIL_",MA,No_UpLimit,-0.4))
    BIN(6);

// IOS测试

RUN_PATTERN(2,1,0,0);

    BIN(7);

// ICCH测试
SET_DPS(1,5.5,V,10,MA);
RUN_PATTERN(2,1,0,0);
if(!DPS_MEASURE(1,R20MA,5,"ICCH",MA,1.6,No_LoLimit))
    BIN(8);

// ICCL测试
SET_DPS(1,5.5,V,19,MA);
SET_INPUT_LEVEL(4.5,0);
SET_OUTPUT_LEVEL(2.5,2.5);
RUN_PATTERN(1,1,0,0);
if(!DPS_MEASURE(1,R20MA,5,"ICCL",MA,4.4,No_LoLimit))
    BIN(8);
```
